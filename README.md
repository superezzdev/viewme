## 𝙰𝚋𝚘𝚞𝚝

𝙲𝚞𝚛𝚛𝚎𝚗𝚝𝚕𝚢 𝙸'𝚖 𝚏𝚘𝚌𝚞𝚜𝚎𝚍 𝚘𝚗: 
- 𝙵𝚞𝚕𝚕-𝚂𝚝𝚊𝚌𝚔 𝙳𝚎𝚟𝚎𝚕𝚘𝚙𝚖𝚎𝚗𝚝 
- 𝙰𝙸-𝚙𝚘𝚠𝚎𝚛𝚎𝚍 𝙰𝚙𝚙𝚕𝚒𝚌𝚊𝚝𝚒𝚘𝚗𝚜 
- 𝙾𝚙𝚎𝚗 𝚂𝚘𝚞𝚛𝚌𝚎 𝙲𝚘𝚗𝚝𝚛𝚒𝚋𝚞𝚝𝚒𝚘𝚗𝚜 
- 𝙳𝚊𝚝𝚊 𝚂𝚝𝚛𝚞𝚌𝚝𝚞𝚛𝚎𝚜 & 𝙰𝚕𝚐𝚘𝚛𝚒𝚝𝚑𝚖𝚜 
- 𝚂𝚢𝚜𝚝𝚎𝚖 𝙳𝚎𝚜𝚒𝚐𝚗 
- 𝙱𝚊𝚌𝚔𝚎𝚗𝚍 𝙴𝚗𝚐𝚒𝚗𝚎𝚎𝚛𝚒𝚗𝚐 <br>
  
𝙸 𝚎𝚗𝚓𝚘𝚢 𝚝𝚞𝚛𝚗𝚒𝚗𝚐 𝚒𝚍𝚎𝚊𝚜 𝚒𝚗𝚝𝚘 𝚙𝚛𝚘𝚍𝚞𝚌𝚝𝚜, 𝚌𝚘𝚗𝚝𝚛𝚒𝚋𝚞𝚝𝚒𝚗𝚐 𝚝𝚘 𝚘𝚙𝚎𝚗 𝚜𝚘𝚞𝚛𝚌𝚎, 𝚊𝚗𝚍 𝚌𝚘𝚗𝚝𝚒𝚗𝚞𝚘𝚞𝚜𝚕𝚢 𝚒𝚖𝚙𝚛𝚘𝚟𝚒𝚗𝚐 𝚖𝚢 𝚎𝚗𝚐𝚒𝚗𝚎𝚎𝚛𝚒𝚗𝚐 𝚜𝚔𝚒𝚕𝚕𝚜. <br> 𝙼𝚢 𝚐𝚘𝚊𝚕 𝚒𝚜 𝚝𝚘 𝚋𝚎𝚌𝚘𝚖𝚎 𝚊 𝚜𝚘𝚏𝚝𝚠𝚊𝚛𝚎 𝚎𝚗𝚐𝚒𝚗𝚎𝚎𝚛 𝚠𝚑𝚘 𝚋𝚞𝚒𝚕𝚍𝚜 𝚒𝚖𝚙𝚊𝚌𝚝𝚏𝚞𝚕 𝚙𝚛𝚘𝚍𝚞𝚌𝚝𝚜 𝚞𝚜𝚎𝚍 𝚋𝚢 𝚖𝚒𝚕𝚕𝚒𝚘𝚗𝚜 𝚘𝚏 𝚙𝚎𝚘𝚙𝚕𝚎.
